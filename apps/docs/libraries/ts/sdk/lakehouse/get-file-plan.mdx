---
title: "Get File Plan"
description: "Get scoped lakehouse file URLs and temporary credentials"
---

<RequestExample>

```ts
const { result, error } = await unprice.lakehouse.getFilePlan({
  projectId: "project_1234567890",
  interval: "30d",
  tables: ["usage"]
});

if (error) {
  console.error(error.message);
  return;
}

console.log(result);
```

</RequestExample>

## Request

<ParamField body="projectId" type="string">
    Optional project ID. Defaults to the authenticated project.
</ParamField>

<ParamField body="customerId" type="string">
    Optional customer ID filter.
</ParamField>

<ParamField body="tables" type="array">
    Optional table list. Supported values include `usage`, `verification`, `metadata`, and `entitlement_snapshot`.
</ParamField>

<ParamField body="interval" type="string">
    File window to inspect. Defaults to `30d`.
</ParamField>
